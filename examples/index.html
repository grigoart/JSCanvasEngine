/*
	Initializing:
		1) Create Canvas instance. Constructor argument: id of canvas on the page;
		2) Create Playground instance. Constructor argument: instance of Canvas;
		3) Playground instance contains InputManager instance, which can be used for work with user inputs
*/

//Create Canvas instance. Constructor argument: id of canvas on the page
const CANVAS = new Canvas("canvas")

//Create Playground instance. Constructor argument: instance of Canvas
const PLAYGROUND = new Playground(CANVAS)

//Playground instance contains InputManager instance, which can be used for work with user inputs
const INPUT_MANAGER = PLAYGROUND.inputManager

//Put text object
PLAYGROUND.putObject(new Text(9, 2, "Hello World"))
